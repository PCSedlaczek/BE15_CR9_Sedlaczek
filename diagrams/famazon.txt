_________________________________

1. User management
  
user
  id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  username VARCHAR(),
  first_name VARCHAR(),
  last_name VARCHAR(),
  password TEXT,
  email VARCHAR(),
  created_at TIMESTAMP,
  modified_at TIMESTAMP,

user_address
  address_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_user_id INT,
  address_line1 VARCHAR(),
  address_line2 VARCHAR(),
  city VARCHAR(),
  zip/postal_code VARCHAR(),
  telephone VARCHAR(),
  mobile VARCHAR(),

user_billing
  billing_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_user_id INT,
  payment_type ENUM,
    [paypal, credit_card, debit_card, invoice]
  provider VARCHAR(),
  account_no INT,
  expiry DATE,

2. Product management

product
  product_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(),
  brand VARCHAR(),
  desc TEXT,
  SKU/barcode VARCHAR(),
  fk_category_id INT,
  fk_inventory_id INT,
  price DECIMAL,
  fk_discount_id INT,
  fk_supplier_id INT,
  fk_shipper_id INT,
  created_at TIMESTAMP,
  modified_at TIMESTAMP,

product_supplier
  supplier_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  supplier_name VARCHAR(),
  supplier_country VARCHAR(),
  supplier_city VARCHAR(),

product_shipper
  shipper_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  shipper_name VARCHAR(),
  shipper_country VARCHAR(),
  shipper_city VARCHAR(),

product_inventory
  inventory_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  quantity INT,
  created_at TIMESTAMP,
  modified_at TIMESTAMP,

product_category
  cat_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(),
  desc TEXT,
  created_at TIMESTAMP,
  modified_at TIMESTAMP,

product_discount
  discount_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(),
  desc TEXT,
  discount_percent DECIMAL,
  active BOOLEAN,
  created_at TIMESTAMP,
  modified_at TIMESTAMP,

3. Shopping process

cart_session
  session_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_user_id INT,
  total DECIMAL,
  created_at TIMESTAMP,
  modified_at TIMESTAMP,

cart_item
  item_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_session_id INT,
  fk_prod_id INT,
  quantity INT,
  created_at TIMESTAMP,
  modified_at TIMESTAMP,

order_items
  items_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_order_id INT,
  fk_prod_id INT,
  quantity INT,
  created_at TIMESTAMP,
  modified_at TIMESTAMP,

order_details
  order_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_user_id INT,
  total DECIMAL,
  fk_invoice_no INT,
  created_at TIMESTAMP,
  modified_at TIMESTAMP,

order_invoice
  invoice_no INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fk_order_id INT,
  amount INT,
  provider VARCHAR(),
  status VARCHAR(),
  created_at TIMESTAMP,
  modified_at TIMESTAMP,

_________________________________

products
  PK prod_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  prod_name VARCHAR(),
  description VARCHAR(),
  price INT,
  type VARCHAR(),
  color VARCHAR(),
  fk_cat_id INT,
  fk_seller_id INT

categories
  PK cat_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  cat_name VARCHAR(),

prod_to_cat
  PK fk_prod_id INT,
  PK fk_cat_id INT

seller
  PK seller_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  seller_name VARCHAR(),
  seller_address VARCHAR(),
  seller_country VARCHAR(),

customer
  PK cust_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  fname VARCHAR(),
  lname VARCHAR(),
  username VARCHAR(),
  email VARCHAR(),
  fk_address VARCHAR(),
  fk_payment VARCHAR(),

prod_to_cust
  PK fk_cust_id INT,
  PK fk_prod_id INT,
  invoice_no INT,
  invoice_price INT

payments
  PK pay_type ENUM [paypal, credit_card, debit_card, invoice]
  card_num INT,
  secure_code INT,
  valid_thru INT

cust_to_pay
  PK fk_cust_id INT,
  PK fk_pay_type VARCHAR(),

shipper
  PK ship_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  ship_name VARCHAR(),
  ship_country VARCHAR(),
  ship_address VARCHAR(),

prod-to-ship
  PK fk_ship_id INT,
  PK fk_prod_id INT
